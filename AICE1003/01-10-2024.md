balance : int = 0
def deposit(ammount : int = 1): int
	balance += ammount
	return balance
def withdraw(ammount : int = 5): int
	if (ammount > balance) {
		return -1
	}
	balance -= ammount
	return balance

deposit()

if (withdraw() == -1): print("Cannot Go Into Debt");
